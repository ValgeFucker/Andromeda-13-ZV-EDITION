## Конфигурация подключения MySQL
## Используется для статистики, сбора отзывов,
## администрирования и внутриигровой библиотеки.

## Следует ли включить SQL? Раскомментируйте для включения.
#SQL_ENABLED

## Сервер, на котором находится база данных MySQL.
## Примеры: localhost, 200.135.5.43, www.mysqldb.com, и т.д.
ADDRESS localhost

## Порт сервера MySQL (по умолчанию 3306).
PORT 3306

## База данных для всех функций SQL, а не только для отзывов.
FEEDBACK_DATABASE feedback

## Префикс, добавляемый к имени каждой таблицы. Для старых баз данных может потребоваться установить значение `erro_`.
## Примечание: это не меняет названия таблиц в самой базе данных, это придётся сделать вручную.
## Например:
##	FEEDBACK_TABLEPREFIX
##	FEEDBACK_TABLEPREFIX SS13_
## Удалите "SS13_", если вы используете стандартную схему (schema file).
FEEDBACK_TABLEPREFIX SS13_

## Имя пользователя/логин для доступа к базе данных.
FEEDBACK_LOGIN username

## Пароль для доступа к базе данных.
FEEDBACK_PASSWORD password

## Время в секундах до таймаута асинхронных запросов
## Установите 0 для бесконечного ожидания
ASYNC_QUERY_TIMEOUT 10

## Время в секундах на выполнение блокирующих запросов до таймаута (slow query timeout)
## Установите 0 для бесконечного ожидания
## Должно быть меньше или равно ASYNC_QUERY_TIMEOUT
BLOCKING_QUERY_TIMEOUT 5

## Минимальное количество SQL-соединений, которые будут поддерживаться в пуле. Установка большего значения на серверах, географически удалённых от базы данных, может улучшить производительность.
POOLING_MIN_SQL_CONNECTIONS 1

## Максимальное количество SQL-соединений с базой данных.
POOLING_MAX_SQL_CONNECTIONS 25

## Максимальное количество одновременных асинхронных запросов, ожидающих результат, прежде чем новые запросы начнут ставиться в очередь.
MAX_CONCURRENT_QUERIES 25
